```python
def _create_sudoku_edges(self):
    edges = []
    for row in range(9):
        for col in range(9):
            src = row * 9 + col
            
            # Row edges
            for k in range(9):
                if k != col:
                    edges.append([src, row*9 + k])
            
            # Column edges
            for k in range(9):
                if k != row:
                    edges.append([src, k*9 + col])
            
            # Box edges (3√ó3)
            box_row, box_col = row // 3, col // 3
            for i in range(box_row*3, (box_row+1)*3):
                for j in range(box_col*3, (box_col+1)*3):
                    if i != row or j != col:
                        edges.append([src, i*9 + j])
    
    return edges
```
